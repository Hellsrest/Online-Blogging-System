    function formatTime($timeDifference)
    {
        $timePeriods = array(
            31536000 => 'year',
            2592000 => 'month',
            86400 => 'day',
            3600 => 'hour',
            60 => 'minute',
            1 => 'second'
        );
        foreach ($timePeriods as $seconds => $period) {
            $numberOfPeriods = floor($timeDifference / $seconds);
            if ($numberOfPeriods > 0) {
                return $numberOfPeriods . ' ' . $period . ($numberOfPeriods > 1 ? 's' : '');
            }
        }
        return 'just now';
    }
